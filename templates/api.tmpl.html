TinyAlias Create Link API


GET https://api.tinyalias.com/create?url={URL}&alias={ALIAS}&password={PASSWORD}&expiration={EXPIRATION}

Example


GET https://api.tinyalias.com/create?url=example.com&alias=example
Response:
{
    "original": "example.com",
    "short": "https://tinyalias.com/example",
    "success": true
}
            

GET https://api.tinyalias.com/create?url=tinyalias.com&alias=tiny&password=abcdef
Response:
{
    "original": "tinyalias.com",
    "password": "abcdef",
    "short": "https://tinyalias.com/tiny",
    "success": true
}
            

GET https://api.tinyalias.com/create?url=api.tinyalias.com&alias=tinyapi&expiration=1539096986
Response:
{
    "expiration": 1539096986,
    "original": "api.tinyalias.com",
    "short": "https://tinyalias.com/tinyapi",
    "success": true
}
